/* crear la base de datos */
create database gestion;

/* crear las tablas */

create table usuario(
	user_id int auto_increment,
	name varchar(43) not null,
	pass varchar(43) not null,
	email varchar(43) not null unique,
	date_time date not null,
	constraint pk_user_id primary key (user_id)
);

create table tarea ( 
	id_tarea int auto_increment, 		
	name varchar(100), 
	description varchar(100), 
	user_id int not null,
	constraint pk_id_tarea primary key (id_tarea), 
	constraint jk_id_usuario foreign key (user_id) references usuario(user_id) 
);
